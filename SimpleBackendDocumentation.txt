        'buy_price', -> سعر الشراء  (رقم)

        'code', -> الكود (رقم)
        'item_unit_id', -> الوحدة  (id)
        'unit_parts_count', -> العدد داخل الوحدة  (رقم)
        ---------------------("NOT_REQUIRED")---------------------
        'img',->(assets/images/uploads/items/',$filename)
        'stock', -> الرصيد الحالي (رقم)
        'starting_stock', رصيد البداية  (رقم) يتم استخدامه اول مرة فقط

        'size_id',
        'color_id',

        'weight',
        'description',
        'manufactory_id', // 'الشركة المنتجة'
        'company_id', // 'الشركة الموزعة'
        'import', // ->boolean
        'importer_id' // 'الشركة المستوردة',
